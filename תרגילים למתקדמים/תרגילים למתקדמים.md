def is_prime(num):
    if num < 2:
        return False
    for i in range(2, int(num ** 0.5) + 1):
        if num % i == 0:
            return False
    return True

def find_primes(n):
    primes = []
    for i in range(2, n + 1):
        if is_prime(i):
            primes.append(i)
    return primes

# code
def convert_temperature(temp, from_unit, to_unit):
    conversions = {
        ("C", "F"): lambda t: (t * 9/5) + 32,
        ("C", "K"): lambda t: t + 273.15,
        ("F", "C"): lambda t: (t - 32) * 5/9,
        ("F", "K"): lambda t: (t - 32) * 5/9 + 273.15,
        ("K", "C"): lambda t: t - 273.15,
        ("K", "F"): lambda t: (t - 273.15) * 9/5 + 32
    }

    if from_unit == to_unit:
        return temp

    key = (from_unit, to_unit)
    if key in conversions:
        return conversions[key](temp)
    else:
        return None

temp = float(input("住 驻专专: "))
from_unit = input(" (C/F/K): ")
to_unit = input(" (C/F/K): ")

result = convert_temperature(temp, from_unit, to_unit)
if result is not None:
    print(f"{temp}掳{from_unit} = {result}掳{to_unit}")
else:
    print("专  转拽")

    positions = []
    for i in range(len(matrix)):

def spiral_matrix(n, m):
    matrix = [[0] * m for _ in range(n)]
    num = 1
    top, bottom = 0, n - 1
    left, right = 0, m - 1

    while top <= bottom and left <= right:
        for j in range(left, right + 1):
            matrix[top][j] = num
            num += 1
        top += 1

        for i in range(top, bottom + 1):
            matrix[i][right] = num
            num += 1
        right -= 1

        if top <= bottom:
            for j in range(right, left - 1, -1):
                matrix[bottom][j] = num
                num += 1
            bottom -= 1

        if left <= right:
            for i in range(bottom, top - 1, -1):
                matrix[i][left] = num
                num += 1
            left += 1

    return matrix

    if len(lst) == 0:
        return []

    matrix = []

        row = []
        for j in range(n):

def is_valid_sudoku(grid):
    # 拽转 砖专转
    for row in grid:
        seen = set()
        for num in row:
            if num != 0:
                if num in seen:
                    return False
                if num < 1 or num > 9:
                    return False
                seen.add(num)

    # 拽转 注转
    for j in range(9):
        seen = set()
        for i in range(9):
            num = grid[i][j]
            if num != 0:
                if num in seen:
                    return False
                seen.add(num)

    # 拽转 专注 3x3
    for box_row in range(0, 9, 3):
        for box_col in range(0, 9, 3):
            seen = set()
            for i in range(box_row, box_row + 3):
                for j in range(box_col, box_col + 3):
                    num = grid[i][j]
                    if num != 0:
                        if num in seen:
                            return False
                        seen.add(num)

    return True

sudoku = [
