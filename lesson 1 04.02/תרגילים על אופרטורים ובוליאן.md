### 驻转专 转专 10:

```python
# code
grade1 = float(input("住 爪 专砖: "))
weight1 = float(input("住 砖拽 专砖: "))
grade2 = float(input("住 爪 砖: "))
weight2 = float(input("住 砖拽 砖: "))
weighted_avg = (grade1 * weight1 + grade2 * weight2) / (weight1 + weight2)
above_60 = weighted_avg >= 60
both_above_70 = grade1 > 70 and grade2 > 70
condition = weighted_avg > 80 or (grade1 > 75 and grade2 > 75)
print(f"爪注 砖拽 : {weighted_avg}")
print(f"注 60? {above_60}")
print(f"砖 爪 注 70? {both_above_70}")
print(f"爪注 注 80  砖 爪 注 75? {condition}")
```
